public class LoggingAspect
{
    public static void LogMethodInvocation(string methodName, LogContext logContext)
    {
        // Log method invocation
        Log.Information($"Method {methodName} called.", logContext);
    }

    public static void LogMethodCompletion(string methodName, LogContext logContext)
    {
        // Log method completion
        Log.Information($"Method {methodName} completed successfully.", logContext);
    }

    public static void LogException(string methodName, LogContext logContext, Exception ex)
    {
        // Log exception
        Log.Error(ex, $"An error occurred in method {methodName}.", logContext);
    }
}
