// YourModel.cs

using System.ComponentModel.DataAnnotations;

namespace DataAnnotationsExampleLib
{
    public class YourModel
    {
        [Required(ErrorMessage = "Name is required.")]
        public string Name { get; set; }
    }
}


---------------------------
static void Main()
        {
            // Create an instance of YourModel
            var yourModel = new DataAnnotationsExampleLib.YourModel();

            // Uncommenting the next line would result in an exception or a validation error
            // yourModel.Name = null;

            // To catch validation errors, you can check the validation state using ValidationContext
            var validationContext = new ValidationContext(yourModel);
            var validationResults = new System.Collections.Generic.List<ValidationResult>();

            bool isValid = Validator.TryValidateObject(yourModel, validationContext, validationResults, true);

            if (!isValid)
            {
                foreach (var validationResult in validationResults)
                {
                    Console.WriteLine(validationResult.ErrorMessage);
                }
            }
            else
            {
                Console.WriteLine("Validation passed. Proceed with further logic.");
            }
        }
