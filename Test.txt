SET SERVEROUTPUT ON
SET LONG 10000
SET LINESIZE 32000

DECLARE
  l_table_name VARCHAR2(30);
  l_sql        VARCHAR2(32767);
  l_clob       CLOB;
BEGIN
  FOR rec IN (SELECT table_name FROM all_tables WHERE owner = 'YOUR_SCHEMA') LOOP
    l_table_name := rec.table_name;
    l_sql := 'SELECT dbms_metadata.get_ddl(''TABLE'', ''' || l_table_name || ''', ''YOUR_SCHEMA'') FROM dual';
    EXECUTE IMMEDIATE l_sql INTO l_clob;
    DBMS_OUTPUT.put_line(l_clob);
  END LOOP;
END;
/

SPOOL export_inserts.sql
@export_inserts.sql
SPOOL OFF
