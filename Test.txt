private readonly ProxyGenerator _generator;
    private readonly LoggingInterceptor _loggingInterceptor;

    public LoggingServiceFactory()
    {
        _generator = new ProxyGenerator();
        _loggingInterceptor = new LoggingInterceptor();
    }

    public ILoggingService CreateLoggingService()
    {
        var loggingService = _generator.CreateInterfaceProxyWithTarget<ILoggingService>(
            new LoggingService(), _loggingInterceptor);
        
        return loggingService;
    }
