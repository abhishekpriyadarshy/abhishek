public class LoggingDecorator
{
    public static void LogMethodInvocation(Action action, LogContext logContext)
    {
        // Log method invocation before executing the action
        LoggingAspect.LogMethodInvocation(action.Method.Name, logContext);
        action();
    }

    public static void LogMethodCompletion(Action action, LogContext logContext)
    {
        try
        {
            action();
            // Log method completion after executing the action
            LoggingAspect.LogMethodCompletion(action.Method.Name, logContext);
        }
        catch (Exception ex)
        {
            // Log exception if an error occurs
            LoggingAspect.LogException(action.Method.Name, logContext, ex);
            throw;
        }
    }
}
